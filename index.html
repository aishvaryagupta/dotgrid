Get Started

Two lines. That's it.

HTML 
<!-- 1. Add the script -->
<script src="dot-matrix-text.js"></script>

<!-- 2. Use it -->
<h1>
  We make the
  <span data-dot-matrix="cycle"
        data-words="future,web,magic"
        data-active-color="#ff5d22"></span>
</h1>
Headline Mode 
<div data-dot-matrix="headline"
     data-lines='["Learn to {ship,build} what","you design."]'
     data-active-color="#ff5d22"
     data-static-color="#f5f5f7"></div>
JS API 
// Auto-initialize all [data-dot-matrix] elements
DotMatrix.init();

// Or create manually with full control
const instance = DotMatrix.create(el, {
  activeColor: '#ff5d22',
  hold: 2500,
  trigger: 'immediate'
});

instance.play();
instance.pause();
instance.destroy();
